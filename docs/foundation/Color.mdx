import { Text, palette, Popover } from "../../src/index";

The colors for a design system is important because it reflects the consistency that a design system maintains.
The color system in palette is quite simple, straight forward and easy to follow.

The Palette component library also exports a palette object which has access to all the color constants mention below.
For Example in order to access the color constant primary, it can be done like this,

```js
palette.primary.primary;
```

> Hover over the color box to get to know the color constant value

export const ColorBox = ({ color, isPrimary, access }) => {
  return (
    <div
      style={{
        display: "flex",
        flexDirection: "row",
        alignItems: "center",
        padding: "10px 0px",
      }}
    >
      <Popover
        placement="left"
        hideArrow
        trigger="hover"
        content={
          <div
            style={{
              width: "200px",
              height: "100px",
              display: "flex",
              flexDirection: "column",
              alignItems: "center",
              justifyContent: "center",
            }}
          >
            <Text size="medium" paddingBottom="5px">
              Color Hex: {color}
            </Text>
            <code
              style={{
                backgroundColor: palette.primary.secondary,
                border: "solid 1px rgb(228, 231, 235)",
                color: "#425a70",
              }}
            >
              {access}
            </code>
          </div>
        }
      >
        <div
          style={{
            height: "40px",
            width: "40px",

            border: "1px solid rgb(228, 231, 235)",

          }}

        <Text paddingX={10} size="medium">
          {isPrimary ? "Primary" : "Secondary"}
        </Text>
        <Text paddingX={10}>{color}</Text>
      </div>
    </div>
  );
};
